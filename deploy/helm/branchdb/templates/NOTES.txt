API サーバーのアクセス方法 (service.type: {{ .Values.apiServer.service.type }}):
{{- if eq .Values.apiServer.service.type "LoadBalancer" }}

  export API_URL=http://$(kubectl -n {{ .Release.Namespace }} get svc {{ include "branchdb.apiServer.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}'):{{ .Values.apiServer.listenPort }}
{{- else if eq .Values.apiServer.service.type "NodePort" }}

  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[?(@.type=="ExternalIP")].address}')
  export NODE_PORT=$(kubectl -n {{ .Release.Namespace }} get svc {{ include "branchdb.apiServer.fullname" . }} -o jsonpath='{.spec.ports[0].nodePort}')
  export API_URL=http://$NODE_IP:$NODE_PORT
{{- else }}

  kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "branchdb.apiServer.fullname" . }} 8080:{{ .Values.apiServer.listenPort }} &
  export API_URL=http://localhost:8080
{{- end }}

  curl $API_URL/health
  curl -X POST $API_URL/branches -H 'Content-Type: application/json' \
    -d '{"name":"feature-x","snapshot_ref":"<snapshot>","ttl_hours":24}'
  open $API_URL/   # Web コンソール
{{- end }}

DatabaseBranch を kubectl で直接作成することもできます:
  kubectl apply -f - <<'EOF'
  apiVersion: branchdb.io/v1alpha1
  kind: DatabaseBranch
  metadata:
    name: feature-x
  spec:
    snapshotRef: <スナップショット名>
    ttlHours: 24
  EOF
  kubectl get databasebranches
