  namespace        : {{ .Release.Namespace }}
  branch namespace : {{ include "branchdb-operator.branchNamespace" . }}
  image            : {{ .Values.image.repository }}:{{ .Values.image.tag | default .Chart.AppVersion }}
  ZFS Agent URL    : {{ .Values.zfsAgent.url | default "(未設定)" }}

状態確認:
  kubectl -n {{ .Release.Namespace }} rollout status deploy/{{ include "branchdb-operator.fullname" . }}
  kubectl -n {{ .Release.Namespace }} logs deploy/{{ include "branchdb-operator.fullname" . }}

{{- if not .Values.zfsAgent.url }}

[警告] zfsAgent.url が未設定です。operator は起動時に ZFSAGENT_URL を必須とするため、
        --set zfsAgent.url=http://<zfs-agent>:9090 を指定して再デプロイしてください。
{{- end }}
{{- if and (not .Values.zfsAgent.token) (not .Values.zfsAgent.existingSecret) }}

[警告] zfsAgent.token / zfsAgent.existingSecret が未設定です。ZFS Agent 認証に失敗します。
        --set zfsAgent.token=<token> または既存 Secret を --set zfsAgent.existingSecret=<name> で指定してください。
{{- end }}

DatabaseBranch を作成してみる:
  kubectl apply -f - <<'EOF'
  apiVersion: branchdb.io/v1alpha1
  kind: DatabaseBranch
  metadata:
    name: feature-x
  spec:
    snapshotRef: <スナップショット名>
    ttlHours: 24
  EOF

  kubectl get databasebranches
